function getWeekNumber(dateStr) {
    // Parse the date string to a Date object
    const parts = dateStr.match(/(\d{2})([A-Z]{3})(\d{4})/);
    const day = parseInt(parts[1], 10);
    const month = new Date(Date.parse(parts[2] + " 1, 2024")).getMonth();
    const year = parseInt(parts[3], 10);
    const date = new Date(year, month, day);

    // Calculate the week number
    const firstJan = new Date(date.getFullYear(), 0, 1);
    const pastDaysOfYear = (date - firstJan) / 86400000 + 1;

    return Math.ceil(pastDaysOfYear / 7);
}

function getNumberOfDutySections() {
    // Check if the "duty_sections" array exists in the db
    if (db && db.duty_sections) {

function createOrUpdateRoster(week_of_the_year, number_of_duty_sections) {
    // Check if "rosters" exists in the db, if not, create it
    if (!db.rosters) {
        db.rosters = {};
    }

    // If the "week_of_the_year" doesn't exist in "rosters", create it
    if (!db.rosters[week_of_the_year]) {
        // Initialize the object for the given week
        db.rosters[week_of_the_year] = {
            starting_dsn: db.duty_sections[0].dsn
        };
    } else {
        // If "rosters" exists, find the last used "starting_dsn" and determine the next duty section
        const last_week = Object.keys(db.rosters).sort((a, b) => b - a)[0];
        const last_starting_dsn = db.rosters[last_week].starting_dsn;

        // Find the index of the last "starting_dsn" in "duty_sections"
        const last_dsn_index = db.duty_sections.findIndex(section => section.dsn === last_starting_dsn);

        // Calculate the index of the next duty section
        const next_dsn_index = (last_dsn_index + 1) % number_of_duty_sections;

        // Update the "starting_dsn" for the current week
        db.rosters[week_of_the_year].starting_dsn = db.duty_sections[next_dsn_index].dsn;
    }

    // Save to local storage
    localStorage.setItem('db', JSON.stringify(db));
}

function ensureStudentStatus() {
    // Check if the "students" array exists in the db
    if (db && db.students) {
        // Iterate over each student in the "students" array
        db.students.forEach(student => {
            // Check if the "status" key does not exist
            if (!student.hasOwnProperty('status')) {
                // Create the "status" key with the value 5
                student.status = 5;
            }
        });

        // Save the updated db back to localStorage
        localStorage.setItem('db', JSON.stringify(db));
    } else {
        console.log('No students found in the database.');
    }
}
